1function [r, Pk] = kalman_filter_lr (r, A, Pk, Q, R_fakt, delta, xl, yl, xr, yr) %#codegen
  2
  3% "kalman_filter_lr" 
  4% 
  5% ----Nomenklatur----
  6%  * r: Zustandsvektor = [y0, phi0, kappa_1, kappa_2, ... , kappa_(n-1)] 
  7%   --> y0: y-Wert des ersten Punktes (x-Wert ist immer 0)
  8%   --> phi0: Anfangssteigung zwischem ersten und zweiten Punkt
  9%   --> kappa_i: Krümmungen an den entsprechenden Punkten (Krümmungsdefinition: 1/R, 
 10%   wobei R der Radius des Kreises ist, auf dem der betrachtete Punkt, sein Vorgänger 
 11%   und sein Nachfolger liegen)
 12%  * Pk: Kovarianzmatrix des Zustands dim[n x n], mit n = Dimension des Zustandsvektors
 13%  * A: State-Transition-Matrix dim[n x n] (ist zurzeit eine Einheitsmatrix, da noch keine 
 14%    Messwerte fuer die Eigenbewegung verfügbar sind)
 15%  * Q: Kovarianz des Zustandsübergangs dim[n x n] (symmetrische Matrix, mit weg von der 
 16%    Diagonalen abnehmenden Eintraegen)
 17%  * R_fakt: Unsicherheit der Messwerte, sollte in der Groessenordnung der Diagonale von Q liegen
 18%  * delta: Abstand der Punkte (delta*n ergibt die Länge des praedizierten Fahrstreifens)
 19%  * xl, yl: Vektoren mit den Messwerten für die linke Spur
 20%  * xr, yr: Vektoren mit den Messwerten für die rechte Spur
 21% 
 22%  ----Grober Ablauf des Algorithmus----
 23%  1. Projektion der Punkte von der Mittellinie nach links bzw. rechts
 24%  2. Fuer jeden Messpunkt: Berechnung des kleinsten Abstands zum aktuell praedizierten Strassenverlauf
 25%  3. Assemblierung der Jakobimatrix fuer die Projektion aus dem Zustandsraum r auf x-y-Koordinaten
 26%  4. Kalman-Filter: Praediktion -> Messwerte einbeziehen -> Update
 27%  5. Zustandsbegrenzungen einbringen
 28
 29
 30
 31%% linke Seitenlinie
 32[xp, yp, phi] = projectPoints(r, delta, 0.38); %Punkte von Mittellinie nach links projizieren
 33P = [xp, yp, phi];
 34D = 10000*ones(numel(xl), 3);
 35
 36% Fuer jeden Messpunkt den nähesten Punkt der aktuellen Praediktion finden 
 37for s=1:numel(r)
 38   for m=1:numel(xl)            
 39        dist_point = sqrt((P(s, 1)-(xl(m)))^2 + (P(s, 2)-(yl(m)))^2);
 40        if dist_point < D(m, 3)
 41            D(m, 1) = s;
 42            D(m, 2) = 0;
 43            D(m, 3) = dist_point;            
 44        end            
 45        if s > 1
 46            [dist_line, lambda, ~] = d_line_point([P(s-1, 1), P(s-1, 2)], [P(s, 1), P(s, 2)], [xl(m), yl(m)]);
 47            if dist_line < D(m, 3) && lambda > 0 && lambda < 1
 48                D(m, 1) = s;
 49                D(m, 2) = lambda;
 50                D(m, 3) = dist_line;              
 51            end
 52        end         
 53    end 
 54end
 55
 56% Messpunkte ausfiltern, die vor dem letzten prädizierten Punkt liegen
 57ind = ((D(:, 1) == numel(r) & D(:, 2) == 0) | D(:, 3) > 0.5);
 58xl(ind) = [];
 59yl(ind) = [];
 60D(ind, :) = [];
 61
 62
 63% Messmatrix, Mess- und Erwartungsvektor bauen
 64Hl = messmatrix_new(P, r, delta, D(:, 1), D(:, 2));
 65zl =  zeros(2*numel(xl), 1);
 66zml = zeros(2*numel(xl), 1);
 67count = 1;
 68for m=1:numel(xl)
 69        zl(count) = xl(m);
 70        zl(count + 1) = yl(m);
 71        if D(m, 2) > 0 
 72            zml(count)     = P(D(m, 1)-1, 1) + D(m, 2) * (P(D(m, 1), 1) - P(D(m, 1)-1, 1));
 73            zml(count + 1) = P(D(m, 1)-1, 2) + D(m, 2) * (P(D(m, 1), 2) - P(D(m, 1)-1, 2));
 74        else
 75            zml(count) = P(D(m, 1), 1);
 76            zml(count + 1) = P(D(m, 1), 2);  
 77        end         
 78        count = count + 2;
 79end
 80
 81
 82
 83%% rechte Seitenlinie
 84[xp, yp, phi] = projectPoints(r, delta, -0.38); %Punkte von Mittellinie nach rechts projizieren
 85P = [xp, yp, phi];
 86D = 10000*ones(numel(xr), 3); 
 87
 88% Fuer jeden Messpunkt den nähesten Punkt der aktuellen Praediktion finden
 89for s=1:numel(r)
 90   for m=1:numel(xr)            
 91        dist_point = sqrt((P(s, 1)-(xr(m)))^2 + (P(s, 2)-(yr(m)))^2);
 92        if dist_point < D(m, 3)
 93            D(m, 1) = s;
 94            D(m, 2) = 0;
 95            D(m, 3) = dist_point;            
 96        end            
 97        if s > 1
 98            [dist_line, lambda, ~] = d_line_point([P(s-1, 1), P(s-1, 2)], [P(s, 1), P(s, 2)], [xr(m), yr(m)]);
 99            if dist_line < D(m, 3) && lambda > 0 && lambda < 1
100                D(m, 1) = s;
101                D(m, 2) = lambda;
102                D(m, 3) = dist_line;              
103            end
104        end         
105    end 
106end
107
108% Messpunkte ausfiltern, die vor dem letzten prädizierten Punkt liegen
109ind = ((D(:, 1) == numel(r) & D(:, 2) == 0) | D(:, 3) > 0.5 );
110xr(ind) = [];
111yr(ind) = [];
112D(ind, :) = [];
113
114% Messmatrix, Mess- und Erwartungsvektor bauen
115Hr = messmatrix_new(P, r, delta, D(:, 1), D(:, 2));
116
117zr =  zeros(2*numel(xr), 1);
118zmr = zeros(2*numel(xr), 1);
119count = 1;
120for m=1:numel(xr)
121        zr(count) = xr(m);
122        zr(count + 1) = yr(m);
123        if D(m, 2) > 0 
124            zmr(count)     = P(D(m, 1)-1, 1) + D(m, 2) * (P(D(m, 1), 1) - P(D(m, 1)-1, 1));
125            zmr(count + 1) = P(D(m, 1)-1, 2) + D(m, 2) * (P(D(m, 1), 2) - P(D(m, 1)-1, 2));
126        else
127            zmr(count) = P(D(m, 1), 1);
128            zmr(count + 1) = P(D(m, 1), 2);  
129        end            
130        count = count + 2;
131end
132
133
134
135%% Linke und rechte Linie kombinieren
136
137H = [Hl; Hr];
138z = [zl; zr];
139zm = [zml; zmr];
140
141
142%% KALMAN FILTER    
143% r = A*r + B*u
144R = eye(2*(numel(xl)+numel(xr))).*R_fakt; 
145
146Pk = A*Pk*A' + Q;
147y_tilde = z - zm;
148K = Pk*H'/(H*Pk*H' + R);
149r = r + K*y_tilde;
150Pk = (eye(numel(r)) - K*H)*Pk;
151
152
153%% Zustandsbegrenzungen
154% Krümmung
155kr = 0.75; % bezogen auf Mittellinie, 0.72 entspricht einem minimalen Innenradius von 1m
156for i=3:numel(r)
157    r(i) = max(-kr, r(i));
158    r(i) = min(kr, r(i));
159end
160
161% y-Wert des ersten Punktes
162y_max = 0.5;
163r(1) =  max(-y_max, min(y_max, r(1)));
164
165% Startwinkel 
166phi_max = 0.78;
167r(2) =  max(-phi_max, min(phi_max, r(2))); % 45°
168
